% Action: Grasp the banana
move(state(middle, onbox, middle, hasnot),
     grasp,
     state(middle, onbox, middle, has)).

% Action: Climb on the box
move(state(P, onfloor, P, H),
     climb,
     state(P, onbox, P, H)).

% Action: Drag the box
move(state(P1, onfloor, P1, H),
     drag(P1, P2),
     state(P2, onfloor, P2, H)).

% Action: Walk to another place
move(state(P1, onfloor, B, H),
     walk(P1, P2),
     state(P2, onfloor, B, H)).

% Goal check: monkey has the banana
canget(state(_, _, _, has)).

% Recursive search: check if goal can be reached
canget(State1) :-
    move(State1, _, State2),
    canget(State2).

% Main program
:- initialization(main).

main :-
    canget(state(atdoor, onfloor, atwindow, hasnot)),
    write('Yes, the monkey can get the banana!'), nl.
